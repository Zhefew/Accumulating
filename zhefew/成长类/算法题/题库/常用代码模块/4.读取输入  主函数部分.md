int main() {
    // 读取输入
    int k;
    cin >> k;
    vector<ListNode*> lists;
    for (int i = 0; i < k; ++i) {
        vector<int> nums;
        int num;
        while (cin >> num) {
            nums.push_back(num);
            if (cin.get() == '\n') break; // 读到换行符结束
        }
        lists.push_back(createList(nums));
    }
    
    // 合并并输出
    Solution solution;
    ListNode* result = solution.mergeKLists(lists);
    printList(result);
    
    return 0;
}
